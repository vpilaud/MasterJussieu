\section{Plan du cours}

\begin{description}
\item[1. Cones polyèdriques] ~
	\begin{itemize}
	\item Définition des cones et propriétés élémentaires
	\item Thm de Minkowski-Weyl pour les cones (preuve par lancer de rayon)
	\item Élimination de Fourier-Motzkin
	\item Polarité
	\end{itemize}
	TD1. 

\medskip
\item[2. Polytopes, polyèdres] ~
	\begin{itemize}
	\item Convexité -- Thm de Carathéodory -- Thm de Radon
	\item Définition des polytopes et polyèdres ($V$ et $H$ descriptions)
	\item Exemples (simplexes, cubes, cross-polytopes, $0/1$-polytopes issus de problèmes d'optimisation combinatoire)
	\item Thm de Minkowski-Weyl pour les polytopes
	\item Polarité
	\end{itemize}
	TD2.

\medskip
\item[3. Faces] ~
	\begin{itemize}
	\item Définitions des faces et du $f$-vecteur, propriétés élémentaires
	\item Treillis des faces d'un polyèdre
	\item Polytopes simples et simpliciaux
	\item Cones et éventails normaux
	\end{itemize}
	TD3.

\medskip
\item[4. Opérations et exemples] ~
	\begin{itemize}
	\item Produit Cartésien
	\item Somme directe
	\item Join
	\item Pyramide
	\item Somme de Minkowski
	\item Projections / sections
	\end{itemize}
	TD4.

\medskip

	\begin{itemize}
	\item Thm de Balinski
	\end{itemize}
	TD5. Thm de Blind-Mani et Kalai (graphes de polytopes simples)
