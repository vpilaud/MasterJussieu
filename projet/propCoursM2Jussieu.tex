\section{Plan du cours}

\begin{description}
\item[Semaine A] ~
\begin{description}
\item[Cours A1. Cones polyèdriques] ~
	\begin{itemize}
	\item Définition des cones et propriétés élémentaires
	\item Thm de Minkowski-Weyl pour les cones (preuve par lancer de rayon)
	\item Élimination de Fourier-Motzkin
	\item Polarité
	\end{itemize}

\smallskip
\item[Cours A2. Polytopes, polyèdres] ~
	\begin{itemize}
	\item Convexité -- Thm de Carathéodory -- Thm de Radon
	\item Définition des polytopes et polyèdres ($V$ et $H$ descriptions)
	\item Exemples (simplexes, cubes, cross-polytopes, $0/1$-polytopes issus de problèmes d'optimisation combinatoire)
	\item Thm de Minkowski-Weyl pour les polytopes
	\item Polarité
	\end{itemize}
	
\smallskip
\item[TD A.] ~
	\begin{itemize}
	\item Convexité (généralisations du thm de Carathéodory, thm de Helly)
	\item Projections (tout polytope est la projection d'un simplexe, tout zonotope est la projection d'un cube, tout polytope centralement symétrique est la projection d'un cross-polytope)
	\item Introduction aux diagrammes de Schlegel
	\end{itemize}
\end{description}

\medskip
\item[Semaine B] ~
\begin{description}
\item[B1. Faces] ~
	\begin{itemize}
	\item Définitions des faces et du $f$-vecteur, propriétés élémentaires
	\item Treillis des faces d'un polyèdre
	\item Polytopes simples et simpliciaux
	\item Cones et éventails normaux
	\end{itemize}

\smallskip
\item[B2. Opérations et exemples] ~
	\begin{itemize}
	\item Produit Cartésien
	\item Somme directe
	\item Join
	\item Pyramide
	\item Somme de Minkowski
	\item Projections / sections
	\end{itemize}

\smallskip
\item[TD B.] Deux familles de polytopes : 
	\begin{itemize}
	\item polytope des stables d'un graphe
	\item polytopes de transport
	\end{itemize}
\end{description}

\medskip
\item[Semaine C] ~

%\smallskip
\item[C1. Relations sur les $f$-vecteurs] ~
	\begin{itemize}
	\item Relation d'Euler (unique relation affine sur le $f$-vecteur des polytopes généraux)
	\item $h$-vecteurs des polytopes simples et simpliciaux
	\item Relations de Dehn-Sommerville
	\end{itemize}

\smallskip
\item[C2. Polytopes extrémaux] ~
	\begin{itemize}
	\item Polytopes cycliques et thm de la borne supérieure
	\item Polytopes empilés et thm de la borne inférieure (preuve de Kalai par rigidité)
	\end{itemize}

\smallskip
\item[TD C.] ~
	\begin{itemize}
	\item Thm de Blind-Mani et Kalai (graphes de polytopes simples)
	\item Non unimodalité du $f$-vecteur
	\item Critère de parité de Gale et $f$-vecteur des polytopes cycliques
	\end{itemize}
\end{description}

\medskip
\item[Semaine D] ~
\begin{description}
\item[D1. Introduction des matroides orientés] ~
	\begin{itemize}
	\item Introduction aux matroides
	\item Arrangements de vecteurs et d'hyperplans
	\item Circuits, cocircuits, chirotope (et leurs axiomes)
	\item Opérations de contraction et deletion
	\item Représentation topologique de Folkmann-Lawrence (sans preuve)
	\end{itemize}

\smallskip
\item[D2. Dualité de Gale] ~
	\begin{itemize}
	\item Orthogonalité dépendances linéaires / évaluations linéaires
	\item Dualité de Gale pour les points
	\item Diagrammes affines
	\item Application : classification des $d$-polytopes à $d+2$ sommets
	\end{itemize}

\smallskip
\item[TD D.] ~
	\begin{itemize}
	\item Zonotopes
	\item Matroides orientés de graphes
	\item Polytopes neighborly non cycliques
	\end{itemize}
\end{description}

\medskip
\item[Semaine E] ~
\begin{description}
\item[E1. Espace de réalisation d'un matroide orienté] ~
	\begin{itemize}
	\item Espaces semi-algébriques
	\item Espace de réalisation d'un polytope
	\item Familles de sommets affinement indépendants
	\end{itemize}

\smallskip
\item[E2. Polytopes de dimension 3 et Thm de Steinitz] ~
	\begin{itemize}
	\item Thm de Tutte pour les graphes planaires par etirement
	\item Relèvement de Maxwell
	\item Réalisation entière et borne sur la taille des coordonnées
	\end{itemize}
	
\smallskip
\item[TD E.] Premières manifestations d'universalité :
	\begin{itemize}
	\item Polytope non rationnel de Perles
	\item Facettes non prescriptibles
	\end{itemize}
\end{description}

\medskip
\item[Semaine F] ~
\begin{description}

\item[F1. Théorèmes d'universalité pour les matroides orientés] ~
	\begin{itemize}
	\item Forme normale de Shor
	\item Equivalence stable
	\item Construction de Von Staudt
	\item Thm d'universalité de Mn\"ev
	\end{itemize}
	
\smallskip
\item[F2. Théorèmes d'universalité pour les polytopes] ~
	\begin{itemize}
	\item Construction de Lawrence
	\item Matroides orientés rigides
	\item Thm d'universalité dimension arbitraire
	\item Thm d'universalité de Richter-Gebert en dimension 4 (sans preuve)
	\end{itemize}

\smallskip
\item[TD F.] Gadgets pour la preuve de Richter-Gebert
\end{description}
\end{description}
